### Number with initial format

export const InputWithFormat = () => {
  const [value, setValue] = React.useState(10);
  return (
    <div style={{ width: 300 }}>
        <FormRow>
            <Input 
                type="number"
                value={value}
                label="Number"
                name="number"
                placeholder="0.00"
                formatInitialValue={(value) => Number(value).toFixed(2)}
                onChange={(e) => setValue(e.target.value)}  
            />
        </FormRow>
    </div>
  );
};

<InputWithFormat />
