### Number with initial format

In general, formatting function executes for initial values.

export const InputWithInitialFormat = () => {
  const [value, setValue] = React.useState(10);
  return (
    <div style={{ width: 300 }}>
        <FormRow>
            <Input 
                type="number"
                value={value}
                label="Number"
                name="number"
                placeholder="0.00"
                format={(value) => Number(value).toFixed(2)}
                onChange={setValue}
            />
        </FormRow>
    </div>
  );
};

<InputWithInitialFormat />

### Event-driven formatted input

There are two events that can used for formatting:
- `blur`
- `input`

#### Blur event 

export const InputWithBlurFormat = () => {
  const [value, setValue] = React.useState(10);
  return (
    <div style={{ width: 300 }}>
        <FormRow>
            <Input 
                type="number"
                value={value}
                label="Number"
                name="number"
                placeholder="0.00"
                formatOnEvent="blur"
                format={(value) => Number(value).toFixed(2)}
                onChange={setValue}
            />
        </FormRow>
    </div>
  );
};

<InputWithBlurFormat />

export const InputWithChangeFormat = () => {
  const [value, setValue] = React.useState('');
  return (
    <div style={{ width: 300 }}>
        <FormRow>

                format={(value) => String(value).toUpperCase()}

  );
};
