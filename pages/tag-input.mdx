## With submitting and removing tags on custom key events

You can set triggers for adding and removing. Default value for addition: `['Enter']`, for removing: `[]`.

Multiple trigger values can be set.

export const KeyCodeTagInput = () => {
    const [values, setValues] = useState([{ id:1, value:'Tag 1' }]);
    const [inputValue, setInputValue] = useState('');

    const addKeyCodes = ['Enter', 'Space', 'Comma'];
    const removeKeyCodes = ['Delete', 'Backspace', 'Escape'];
    return (
      <div style={{ width: 300 }}>
          <TagInput
            id="KeyCodeTagInput"
            tags={values}
            label="Tags"
            placeholder="Enter your tags here"
            addKeyCodes={addKeyCodes}
            removeKeyCodes={removeKeyCodes}
            onChange={setValues}
            inputValue={inputValue}
            setInputValue={setInputValue}
        />
      </div>
    )
}

## With validation and clipboard handling

Tag input with validation and items should be unique and separated by *,*.

Try to copy and paste:

> `name1@mail.com, name1@mail.com, name2mail.com, name3@mail.com`

export const TagInputWithValidation = () => {
  const [values, setValues] = useState([{ id:1, value:'testmail@macpaw.com' }]);
  const [inputValue, setInputValue] = useState('');

  const [error, setError] = useState('');
  const validateEmail = (tag) => {
    const emailRegex = /^[\w-\.]+@([\w-]+\.)+[\w-]{2,14}$/g;
    const isValid = emailRegex.test(tag);
    if(!isValid) setError('Please enter a valid email address');
    return isValid;
  };
  const onChangeInput = () => setError('');
  const handleChange = (nextValues) => {
    setValues(nextValues);
    const hasWrongValues = nextValues.some(({color})=>color === 'warning');
    if(error && !hasWrongValues) setError('');
  }
  return (
    <div style={{ width: 300 }}>
      <TagInput
          id="TagInputWithValidation"
          tags={values}
          error={error}
          label="Email list"
          placeholder="Enter your emails here"
          clipboardSeparator=","
          isUnique
          isHandleClipboard
          validate={validateEmail}
          onChange={handleChange}
          onValueChange={onChangeInput}
          inputValue={inputValue}
          setInputValue={setInputValue}
      />
    </div>
  );
};
