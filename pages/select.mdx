## Multiselect

export const MultiselectExample = () => {
  const [selected, setSelected] = useState([1, 2, 3]);
  const items = [
    { value: 'cmm', label: 'CleanMyMac X' },
    { value: 'gmn', label: 'Gemini' },
    { value: 'cmd', label: 'CleanMyDrive' },
    { value: 'stp', label: 'Setapp' },
    { value: 'lst', label: 'Listen' },
  ];
  return (
    <React.Fragment>
      <div style={{ marginBottom: 16, maxWidth: 300 }}>
        <Multiselect
          placeholder='Select a product'
          onChange={setSelected}
          errorMessage='Please select a product'
          label='With Error'
        >
          {items.map((i) => (
            <option
              key={i.value}
              value={i.value}
              selected={selected.includes(i.value)}
            >
              {i.label}
            </option>
          ))}
        </Multiselect>
      </div>
    </React.Fragment>
  );
};

<MultiselectExample />
