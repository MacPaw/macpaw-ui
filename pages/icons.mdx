import * as icons from '../src/Icons/jsx';

## Icons

export const Grid = () => {
    const iconKeys = Object.keys(icons).sort().filter((key) => !key.startsWith('Payment'));
    return iconKeys.map((iconKey, index) => {
        const Component = icons[iconKey];
        return (
            <Component key={index} title={iconKey} />
        );
    });
}

<div style={{display: 'grid', columnGap: 24, rowGap: 24, gridTemplateColumns: 'repeat(auto-fill, minmax(24px, 1fr))', alignItems: 'center', justifyItems: 'center'}}>
    <Grid />
</div>
